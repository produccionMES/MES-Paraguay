from werkzeug.security import generate_password_hash, check_password_hash
from datetime import datetime
from zoneinfo import ZoneInfo

app = Flask(__name__)

db = SQLAlchemy(app)

login_manager = LoginManager(app)
login_manager.login_view = 'login'

# ==========================================
# 1. MODELOS DE BASE DE DATOS (TABLAS)
# ==========================================

class Usuario(UserMixin, db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(50), unique=True, nullable=False)
    password = db.Column(db.String(250), nullable=False)
    rol = db.Column(db.String(20), nullable=False, default='operario')

class Rack(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    nombre = db.Column(db.String(50), nullable=False, unique=True)
    niveles = db.Column(db.Integer, nullable=False)
    posiciones = db.Column(db.Integer, nullable=False)
    ubicaciones = db.relationship('Ubicacion', backref='rack', lazy=True, cascade="all, delete-orphan")

class Ubicacion(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    rack_id = db.Column(db.Integer, db.ForeignKey('rack.id'), nullable=False)
    nivel = db.Column(db.Integer, nullable=False)
    posicion = db.Column(db.Integer, nullable=False)
    codigo_unico = db.Column(db.String(50), unique=True, nullable=False)
    items = db.relationship('Item', backref='ubicacion', lazy=True, cascade="all, delete-orphan")

class Producto(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    sku = db.Column(db.String(50), unique=True, nullable=False)
    ean = db.Column(db.String(50), unique=True, nullable=True)
    descripcion = db.Column(db.String(150), nullable=False)
    items_en_stock = db.relationship('Item', backref='producto_detalle', lazy=True, cascade="all, delete-orphan")

class Item(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    ubicacion_id = db.Column(db.Integer, db.ForeignKey('ubicacion.id'), nullable=False)
    producto_id = db.Column(db.Integer, db.ForeignKey('producto.id'), nullable=False)
    sub_ubicacion = db.Column(db.String(100), default="General")
    cantidad = db.Column(db.Integer, nullable=False)

# Tablas de Historial
class HistorialAjuste(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    fecha = db.Column(db.DateTime, default=datetime.now)
    sku = db.Column(db.String(50), nullable=False)
    descripcion = db.Column(db.String(150), nullable=False)
    cantidad_anterior = db.Column(db.Integer, nullable=False)
    cantidad_nueva = db.Column(db.Integer, nullable=False)
    motivo = db.Column(db.String(200), nullable=False)
    ubicacion = db.Column(db.String(100), nullable=False)
    usuario = db.Column(db.String(50), nullable=False)

class HistorialDespacho(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    fecha = db.Column(db.DateTime, default=datetime.now)
    sku = db.Column(db.String(50), nullable=False)
    descripcion = db.Column(db.String(150), nullable=False)
    cantidad = db.Column(db.Integer, nullable=False)
    transporte = db.Column(db.String(100), nullable=False)
    origen = db.Column(db.String(100), nullable=False)
    usuario = db.Column(db.String(50), nullable=False, default='Desconocido')

@app.route('/usuarios')
@login_required
def gestionar_usuarios():
    if current_user.rol != 'admin':
        flash('⚠️ Acceso denegado.')
        return redirect(url_for('logistica'))
    todos_los_usuarios = Usuario.query.all()
    return render_template('usuarios.html', usuarios=todos_los_usuarios)

@app.route('/usuarios/eliminar/<int:id>', methods=['POST'])
@login_required
def eliminar_usuario(id):
    if current_user.rol != 'admin': return redirect(url_for('logistica'))
    user = Usuario.query.get_or_404(id)
    if user.id == current_user.id:
        flash('❌ No puedes eliminarte a ti mismo.')
    else:
        db.session.delete(user)
        db.session.commit()
        flash(f'✅ Usuario {user.username} eliminado.')
    return redirect(url_for('gestionar_usuarios'))

@app.route('/usuarios/reset_password/<int:id>', methods=['POST'])
@login_required
def reset_password(id):
    if current_user.rol != 'admin': return redirect(url_for('logistica'))
    user = Usuario.query.get_or_404(id)
    nueva_pass = request.form.get('nueva_password')
    if nueva_pass:
        user.password = generate_password_hash(nueva_pass)
        db.session.commit()
        flash(f'✅ Contraseña de {user.username} actualizada.')
    return redirect(url_for('gestionar_usuarios'))

@app.route('/agregar_item/<int:ubicacion_id>', methods=['POST'])
@login_required
def agregar_item(ubicacion_id):
    if current_user.rol not in ['admin', 'stock']:
        flash('⚠️ No tienes permiso para agregar mercadería.')
        return redirect(request.referrer)

    producto_id = int(request.form.get('producto_id'))
    cantidad = int(request.form.get('cantidad'))
    sub_ubicacion = request.form.get('sub_ubicacion', '').strip()
    if not sub_ubicacion:
        sub_ubicacion = "General"

    # Verificamos si ya existe el mismo producto en la misma posición y LPN
    item_existente = Item.query.filter_by(
        ubicacion_id=ubicacion_id, 
        producto_id=producto_id, 
        sub_ubicacion=sub_ubicacion
    ).first()

    if item_existente:
        item_existente.cantidad += cantidad
        flash(f'✅ Stock actualizado: se sumaron {cantidad} unidades.')
    else:
        nuevo_item = Item(
            ubicacion_id=ubicacion_id,
            producto_id=producto_id,
            cantidad=cantidad,
            sub_ubicacion=sub_ubicacion
        )
        db.session.add(nuevo_item)
        flash('✅ Nuevo producto ingresado en la posición.')

    db.session.commit()
    return redirect(request.referrer)

@app.route('/ajustar_stock/<int:item_id>', methods=['POST'])
@login_required
def ajustar_stock(item_id):
    if current_user.rol not in ['admin', 'stock']:
        flash('⚠️ Acceso denegado.')
        return redirect(request.referrer)

    item = Item.query.get_or_404(item_id) 
    nueva_cantidad = int(request.form.get('nueva_cantidad'))
    motivo = request.form.get('motivo')
    cantidad_anterior = item.cantidad
    
    hora_argentina = datetime.now(ZoneInfo('America/Argentina/Buenos_Aires')).replace(tzinfo=None)
    
    # Historial de Auditoría
    registro_ajuste = HistorialAjuste(
        fecha=hora_argentina,
        sku=item.producto_detalle.sku,
        descripcion=item.producto_detalle.descripcion,
        cantidad_anterior=cantidad_anterior,
        cantidad_nueva=nueva_cantidad,
        motivo=motivo,
        ubicacion=item.ubicacion.codigo_unico,
        usuario=current_user.username
    )
    
    # Historial General (Para que figure el movimiento)
    diferencia = nueva_cantidad - cantidad_anterior
    tipo = "AJUSTE POSITIVO" if diferencia > 0 else "AJUSTE NEGATIVO"
    registro_general = HistorialDespacho(
        fecha=hora_argentina,
        sku=item.producto_detalle.sku,
        descripcion=item.producto_detalle.descripcion,
        cantidad=abs(diferencia),
        transporte=f"🔧 {tipo}: {motivo}",
        origen=item.ubicacion.codigo_unico,
        usuario=current_user.username
    )
    
    item.cantidad = nueva_cantidad
    if item.cantidad <= 0:
        db.session.delete(item) # Limpia la ubicación si queda en 0
        
    db.session.add(registro_ajuste)
    db.session.add(registro_general)
    db.session.commit()
    
    flash('✅ Inventario ajustado correctamente.')
    return redirect(request.referrer)

@app.route('/despachos', methods=['GET', 'POST'])
@login_required
def despachos():
    sku_buscado = request.args.get('sku', '').strip()
    posicion_filtro = request.args.get('posicion_filtro', '').strip()
    resultados = []

    if sku_buscado:
        query = Item.query.join(Producto).filter(
            (Producto.sku == sku_buscado) | (Producto.ean == sku_buscado)
        )
        if posicion_filtro:
            query = query.join(Ubicacion).filter(Ubicacion.codigo_unico == posicion_filtro)
            
        resultados = query.all()

    return render_template('despachos.html', 
                           resultados=resultados, 
                           sku=sku_buscado, 
                           posicion_filtro=posicion_filtro)

@app.route('/descontar_item/<int:item_id>', methods=['POST'])
@login_required
def descontar_item(item_id):
    item = Item.query.get_or_404(item_id)
    cantidad_a_descontar = int(request.form.get('cantidad_descontar'))
    transporte = request.form.get('transporte')

    if cantidad_a_descontar > item.cantidad:
        flash('⚠️ Error: No puedes sacar más stock del que existe.')
        return redirect(request.referrer)

    hora_argentina = datetime.now(ZoneInfo('America/Argentina/Buenos_Aires')).replace(tzinfo=None)

    # Registro del despacho en el historial
    nuevo_despacho = HistorialDespacho(
        fecha=hora_argentina,
        sku=item.producto_detalle.sku,
        descripcion=item.producto_detalle.descripcion,
        cantidad=cantidad_a_descontar,
        transporte=transporte,
        origen=item.ubicacion.codigo_unico,
        usuario=current_user.username
    )
    db.session.add(nuevo_despacho)

    # Restar stock real
    item.cantidad -= cantidad_a_descontar

    if item.cantidad <= 0:
        db.session.delete(item) # Libera la posición si queda vacía

    db.session.commit()
    flash(f'✅ ¡Despacho exitoso! Se restaron {cantidad_a_descontar} unidades.')
    return redirect(request.referrer)
